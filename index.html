Cleaning and create corpus

Cleaning the 3 data and creates a corpus of character vectors

newscorpus <- Corpus(VectorSource(news))
blogscorpus <- Corpus(VectorSource(blogs))
twittercorpus <- Corpus(VectorSource(twitter))
# Remplacement de «/», «@» et «|» avec espace:
toSpace <- content_transformer(function (x , pattern ) gsub(pattern, " ", x))
newscorpus <- tm_map(newscorpus, toSpace, "/")
newscorpus <- tm_map(newscorpus, toSpace, "@")
newscorpus <- tm_map(newscorpus, toSpace, "\\|")

blogscorpus <- tm_map(blogscorpus, toSpace, "/")
blogscorpus <- tm_map(blogscorpus, toSpace, "@")
blogscorpus <- tm_map(blogscorpus, toSpace, "\\|")

twittercorpus <- tm_map(twittercorpus, toSpace, "/")
twittercorpus <- tm_map(twittercorpus, toSpace, "@")
twittercorpus <- tm_map(twittercorpus, toSpace, "\\|")
# Convert the text to lower case
newscorpus <- tm_map(newscorpus, content_transformer(tolower))
blogscorpus <- tm_map(blogscorpus, content_transformer(tolower))
twittercorpus <- tm_map(twittercorpus, content_transformer(tolower))
# Remove numbers
newscorpus <- tm_map(newscorpus, removeNumbers)
blogscorpus <- tm_map(blogscorpus, removeNumbers)
twittercorpus <- tm_map(twittercorpus, removeNumbers)
# Remove punctuations
newscorpus <- tm_map(newscorpus, removePunctuation)
blogscorpus <- tm_map(blogscorpus, removePunctuation)
twittercorpus <- tm_map(twittercorpus, removePunctuation)
# Eliminate extra white spaces
newscorpus <- tm_map(newscorpus, stripWhitespace)
blogscorpus <- tm_map(blogscorpus, stripWhitespace)
twittercorpus <- tm_map(twittercorpus, stripWhitespace)

Identification of bigrams and trigrams of the 3 texts

The identification of bigrams and trigrams of the 3 texts will make it possible to see which are the most common word associations

# bigrammes et trigrammes
  #NEWS
  library(dplyr)
  library(tidytext)
  
  news_df <- tibble(newscorpus2$content)
  
  news_bigrams <- news_df %>%
    unnest_tokens(bigram, newscorpus2$content, token = "ngrams", n = 2)
  count_news_bigrams<-news_bigrams %>%
    count(bigram, sort = TRUE)
  
  news_trigrams <- news_df %>%
    unnest_tokens(trigram, newscorpus2$content, token = "ngrams", n = 3)
  count_news_trigrams<-news_trigrams %>%
    count(trigram, sort = TRUE)
  
  #BLOGS
  
  blogs_df <- tibble(blogscorpus2$content)
  
  blogs_bigrams <- blogs_df %>%
    unnest_tokens(bigram, blogscorpus2$content, token = "ngrams", n = 2)
  count_blogs_bigrams<-blogs_bigrams %>%
    count(bigram, sort = TRUE)
  
  blogs_trigrams <- blogs_df %>%
    unnest_tokens(trigram, blogscorpus2$content, token = "ngrams", n = 3)
  count_blogs_trigrams<-blogs_trigrams %>%
    count(trigram, sort = TRUE)
